name()  {
  "John"
}
    
surname() {
  "Doe"
}
    
fullname(first_name, sep, last_name) {
  first_name + sep + last_name
}

name()
surname()

fullname(name(), " ", surname())
fullname("a", " ", "b")
fullname(name(), name(), surname())



parameters  ::= parameters:rest COMMA STRING_LITERAL:sl   {: RESULT = rest + ", " + "\"" + sl + "\""; :}
                |STRING_LITERAL:sl {:RESULT = "\"" + sl + "\"";:}
                |parameters:rest COMMA nested_function_call:fc   {: RESULT = rest + ", " + fc; :}
                | nested_function_call:fc {:RESULT = fc;:}
                ;
